01 Hermes 自我进化系统
整个系统的"学习大脑"——让 StoryForge AI 不再每次从零开始，而是能从每次生成中自动总结经验、沉淀技能，越用越懂作者的写作风格。贯穿了从后端核心管线到前端展示的完整链路。
backend/core/workflow_optimization.py
Core · Python
Hermes 学习循环的核心实现，包含四条管线：TraceAggregator 聚合每次章节生成的完整执行轨迹（Critic 评分变化、Revise 修改记录、最终质量指标）；FeedbackCollector 从 Critic 低分项、Guardian 违规记录、用户手动修改三个来源采集反馈信号；ExperienceExtractor 利用 LLM 分析问题根因（是设定冲突？语气漂移？还是情节断裂？）；SkillDistiller 将分析结果蒸馏为四类结构化技能（character_style / writing_style / plot_helper / user_preference），按置信度分级自动注册。
TraceAggregator FeedbackCollector ExperienceExtractor SkillDistiller 置信度分级
# TraceAggregator — 聚合执行轨迹 class TraceAggregator: def aggregate(self, run_id: str) -> ExecutionTrace: return ExecutionTrace( critic_scores=self._collect_critic_history(run_id), revise_records=self._collect_revise_records(run_id), user_changes=self._detect_user_modifications(run_id), ) # FeedbackCollector — 多源信号采集 class FeedbackCollector: def collect(self, trace: ExecutionTrace) -> list[FeedbackSignal]: signals = [] if trace.has_low_score(): # Critic 低分 signals.append(FeedbackSignal('critic_low', trace.low_score_items)) if trace.has_guardrail_violations(): # Guardian 违规 signals.append(FeedbackSignal('guardrail', trace.guardrail_errors)) if trace.has_user_edits(): # 用户手动修改 signals.append(FeedbackSignal('user_edit', trace.user_changes)) return signals
// ExperienceExtractor → SkillDistiller 管线 class ExperienceExtractor: def analyze(self, signal: FeedbackSignal) -> WritingExperience: root_cause = self.llm.analyze( # LLM 分析根因 problem=signal.evidence, context=signal.chapter_context ) return WritingExperience( problem_type=root_cause.type, root_cause=root_cause.description, suggestion=root_cause.suggestion, confidence=root_cause.confidence ) class SkillDistiller: def distill(self, experience: WritingExperience) -> Skill: if experience.confidence >= 0.8: # 高置信度 → 直接生效 strength = 0.7 elif experience.confidence >= 0.5: # 中等 → 低强度试用 strength = 0.3 else: # 低置信度 → 标记 draft return Skill.as_draft(experience) return Skill(experience.problem_type, strength)
backend/core/skill_runtime/skill_assembler.py
Core · Python
SkillAssembler — 是 Hermes 学习循环的"消费端"。Writer 每章生成前被调用，根据当前章节的涉及角色、剧情阶段（开端/发展/高潮/收束）、场景类型（对话/战斗/描写/内心独白）三维特征，从 ChromaDB 语义检索 top-5 技能，按置信度排序后动态注入 Writer 的 system prompt。同时负责将预置的名家文风（鲁迅、刘慈欣等）和历史沉淀的经验技能统一装配。
语义检索 ChromaDB 动态注入 多维特征 技能装配
# SkillAssembler — 上下文感知的技能动态注入 class SkillAssembler: def assemble(self, ctx: ChapterContext) -> list[Skill]: # 1. 构建查询：结合角色、剧情阶段、场景类型 query = self._build_query(ctx.characters, ctx.plot_stage, ctx.scene_type) # 2. 向量检索 top-5 技能 candidates = self.vector_db.similarity_search(query, k=5) # 3. 低质量技能过滤 + 按置信度排序 return [s for s in sorted(candidates, key=lambda x: x.confidence, reverse=True) if s.confidence >= 0.5]

// smoothScrollTo — 自定义高品质翻页滚动 function smoothScrollTo(container: HTMLElement, targetY: number) { const start = container.scrollTop const distance = targetY - start const startTime = performance.now() const duration = 900 // 900ms 高品质手感 function step(time: number) { const elapsed = time - startTime const p = Math.min(elapsed / duration, 1) const eased = p < 0.5 ? 4 * p * p * p : 1 - Math.pow(-2 * p + 2, 3) / 2 container.scrollTop = start + distance * eased if (p < 1) requestAnimationFrame(step) } requestAnimationFrame(step) }

封装 IntersectionObserver 的通用 Hook。支持自定义 threshold 和 rootMargin。初始化时自动扫描已可见元素，防止首屏内容空白。提供批量观察 initScrollReveal 和单元素 useScrollReveal 两种使用方式。Landing 页面使用 threshold=0.35 确保翻页到位后才弹入。

// useScrollReveal — 核心实现 export function initScrollReveal(selector = '.reveal', options?: IntersectionObserverInit) { const observer = new IntersectionObserver( (entries) => { entries.forEach((entry) => { if (entry.isIntersecting) { entry.target.classList.add('visible') // 翻出后添加 visible observer.unobserve(entry.target) } }) }, { threshold: 0.35, ...options } ) elements.forEach((el) => observer.observe(el)) }
