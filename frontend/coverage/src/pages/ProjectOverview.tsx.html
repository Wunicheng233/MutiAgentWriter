  const stepIndex = flowSteps.findIndex(step => step.key === stepKey)

  if (project.status === 'draft') {
    if (stepKey === 'idea') return 'active'
    return 'idle'
  }
  if (stepIndex < currentIndex) return 'done'

  return 'idle'
